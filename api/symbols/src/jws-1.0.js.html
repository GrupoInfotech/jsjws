 17 function _getSignatureInputByString(sHead, sPayload) {
 18     return stob64u(sHead) + "." + stob64u(sPayload);
 19 }
 20 
 21 function _getHashBySignatureInput(sSignatureInput, sHashAlg) {
 22     var hashfunc = _RSASIGN_HASHHEXFUNC[sHashAlg];
 23     if (hashfunc == null) throw "hash function not defined in jsrsasign: " + sHashAlg;
 24     return hashfunc(sSignatureInput);
 25 }

 27 function _jws_verifySignature(sHead, sPayload, hSig, hN, hE) {
 28     var sSignatureInput = _getSignatureInputByString(sHead, sPayload);
 29     var biSig = parseBigInt(hSig, 16);
 30     return _rsasign_verifySignatureWithArgs(sSignatureInput, biSig, hN, hE);
 31 }

 33 /**
 34  * get Encoed Signature Value from JWS string.<br/>
 35  * @name getEncodedSignatureValueFromJWS
 36  * @memberOf JWS
 37  * @function
 38  * @param {String} sJWS JWS signature string to be verified
 39  * @return {String} string of Encoded Signature Value 
 40  * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
 41  */
 42 function _jws_getEncodedSignatureValueFromJWS(sJWS) {
 43     if (sJWS.match(/^[^.]+\.[^.]+\.([^.]+)$/) == null) {
 44 	throw "JWS signature is not a form of 'Head.Payload.SigValue'.";
 45     }
 46     return RegExp.$1;
 47 }

 49 /**
 50  * verify JWS signature by RSA public key.<br/>
 51  * This only supports "RS256" and "RS512" algorithm.
 52  * @name verifyJWSByNE
 53  * @memberOf JWS
 54  * @function
 55  * @param {String} sJWS JWS signature string to be verified
 56  * @param {String} hN hexadecimal string for modulus of RSA public key
 57  * @param {String} hE hexadecimal string for public exponent of RSA public key
 58  * @return {String} returns 1 when JWS signature is valid, otherwise returns 0
 59  * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
 60  * @throws if JWS Header is a malformed JSON string.
 61  */
 62 function _jws_verifyJWSByNE(sJWS, hN, hE) {
 63     if (sJWS.match(/^([^.]+)\.([^.]+)\.([^.]+)$/) == null) {
 64 	throw "JWS signature is not a form of 'Head.Payload.SigValue'.";
 65     }
 66     var b6Head = RegExp.$1;
 67     var b6Payload = RegExp.$2;
 68     var b6SigVal = RegExp.$3;
 69     var sSI = b6Head + "." + b6Payload;
 70     this.parsedJWS = {};
 71     this.parsedJWS.headB64U = b6Head;
 72     this.parsedJWS.payloadB64U = b6Payload;
 73     this.parsedJWS.sigvalB64U = b6SigVal;
 74     this.parsedJWS.si = sSI;
 75 
 76     var hSigVal = b64utohex(b6SigVal);
 77     var biSigVal = parseBigInt(hSigVal, 16);
 78     this.parsedJWS.sigvalH = hSigVal;
 79 
 80     var sHead = b64utos(b6Head);
 81     var sPayload = b64utos(b6Payload);
 82     this.parsedJWS.headS = sHead;
 83     this.parsedJWS.payloadS = sPayload;
 84 
 85     if (! this.isSafeJSONString(sHead)) throw "malformed JSON string for JWS Head: " + sHead;
 86 
 87     return _rsasign_verifySignatureWithArgs(sSI, biSigVal, hN, hE);    
 88 }

 92 function _jws_getHashAlgFromHead(sHead) {
 93     var sHeadParsed = jsonParse(sHead);
 94     var sigAlg = sHeadParsed["alg"];
 95     var hashAlg = "";
 96 
 97     if (sigAlg != "RS256" && sigAlg != "RS512")
 98 	throw "JWS signature algorithm not supported: " + sigAlg;
 99     if (sigAlg == "RS256") hashAlg = "sha256";
100     if (sigAlg == "RS512") hashAlg = "sha512";
101     return hashAlg;
102 }

104 function _jws_generateSignatureValueBySI_NED(sHead, sPayload, sSI, hN, hE, hD) {
105     var rsa = new RSAKey();
106     rsa.setPrivate(hN, hE, hD);
107 
108     var hashAlg = _jws_getHashAlgFromHead(sHead);
109     var sigValue = rsa.signString(sSI, hashAlg);
110     return sigValue;
111 }
112 
113 function _jws_generateSignatureValueByNED(sHead, sPayload, hN, hE, hD) {
114     var sSI = _getSignatureInputByString(sHead, sPayload);
115     return _jws_generateSignatureValueBySI_NED(sHead, sPayload, sSI, hN, hE, hD);
116 }

118 /**
119  * generate JWS signature by Header, Payload and a RSA private key.<br/>
120  * This only supports "RS256" and "RS512" algorithm.
121  * @name generateJWSByNED
122  * @memberOf JWS
123  * @function
124  * @param {String} sHead string of JWS Header
125  * @param {String} sPayload string of JWS Payload
126  * @param {String} hN hexadecimal string for modulus of RSA public key
127  * @param {String} hE hexadecimal string for public exponent of RSA public key
128  * @param {String} hD hexadecimal string for private exponent of RSA private key
129  * @return {String} JWS signature string
130  * @throws if sHead is a malformed JSON string.
131  * @throws if supported signature algorithm was not specified in JSON Header.
132  */
133 function _jws_generateJWSByNED(sHead, sPayload, hN, hE, hD) {
134     if (! this.isSafeJSONString(sHead)) throw "JWS Head is not safe JSON string: " + sHead;
135     var sSI = _getSignatureInputByString(sHead, sPayload);
136     var hSigValue = _jws_generateSignatureValueBySI_NED(sHead, sPayload, sSI, hN, hE, hD);
137     var b64SigValue = hextob64u(hSigValue);
138     return sSI + "." + b64SigValue;
139 }

141 /**
142  * check whether a String "s" is a safe JSON string or not.<br/>
143  * If a String "s" is a malformed JSON string or an other object type
144  * this returns 0, otherwise this returns 1.
145  * @name isSafeJSONString
146  * @memberOf JWS
147  * @function
148  * @param {String} s JSON string
149  * @return {Number} 1 or 0
150  */
151 function _jws_isSafeJSONString(s) {
152   var o = null;
153   try {
154     o = jsonParse(s);
155     if (typeof o != "object") return 0;
156     if (o.constructor === Array) return 0;
157     return 1;
158   } catch (ex) {
159     return 0;
160   }
161 }
