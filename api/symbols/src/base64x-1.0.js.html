 45 // ==== string / byte array ================================
 46 /**
 47  * convert a string to an array of character codes
 48  * @param {String} s
 49  * @return {Array of Numbers} 
 50  */
 51 function stoBA(s) {
 52     var a = new Array();
 53     for (var i = 0; i < s.length; i++) {
 54 	a[i] = s.charCodeAt(i);
 55     }
 56     return a;
 57 }
 58 
 59 /**
 60  * convert an array of character codes to a string
 61  * @param {Array of Numbers} a array of character codes
 62  * @return {String} s
 63  */
 64 function BAtos(a) {
 65     var s = "";
 66     for (var i = 0; i < a.length; i++) {
 67 	s = s + String.fromCharCode(a[i]);
 68     }
 69     return s;
 70 }

 72 // ==== byte array / hex ================================
 73 /**
 74  * convert an array of bytes(Number) to hexadecimal string.<br/>
 75  * @param {Array of Numbers} a array of bytes
 76  * @return {String} hexadecimal string
 77  */
 78 function BAtohex(a) {
 79     var s = "";
 80     for (var i = 0; i < a.length; i++) {
 81 	var hex1 = a[i].toString(16);
 82 	if (hex1.length == 1) hex1 = "0" + hex1;
 83 	s = s + hex1;
 84     }
 85     return s;
 86 }

 88 // ==== string / hex ================================
 89 /**
 90  * convert a ASCII string to a hexadecimal string of ASCII codes.<br/>
 91  * NOTE: This can't be used for non ASCII characters.
 92  * @param {s} s ASCII string
 93  * @return {String} hexadecimal string
 94  */
 95 function stohex(s) {
 96     return BAtohex(stoBA(s));
 97 }

 99 // ==== string / base64 ================================
100 /**
101  * convert a ASCII string to a Base64 encoded string.<br/>
102  * NOTE: This can't be used for non ASCII characters.
103  * @param {s} s ASCII string
104  * @return {String} Base64 encoded string
105  */
106 function stob64(s) {
107     return hex2b64(stohex(s));

111 /**
112  * convert a ASCII string to a Base64URL encoded string.<br/>

121 /**
122  * convert a Base64URL encoded string to a ASCII string.<br/>

131 // ==== base64 / base64url ================================
132 /**
133  * convert a Base64 encoded string to a Base64URL encoded string.<br/>
134  * Example: "ab+c3f/==" → "ab-c3f_"
135  * @param {String} s Base64 encoded string
136  * @return {String} Base64URL encoded string
137  */
138 function b64tob64u(s) {
139     s = s.replace(/\=/g, "");
140     s = s.replace(/\+/g, "-");
141     s = s.replace(/\//g, "_");
142     return s;
143 }
144 
145 /**
146  * convert a Base64URL encoded string to a Base64 encoded string.<br/>
147  * Example: "ab-c3f_" → "ab+c3f/=="
148  * @param {String} s Base64URL encoded string
149  * @return {String} Base64 encoded string
150  */
151 function b64utob64(s) {
152     if (s.length % 3 == 1) s = s + "==";
153     if (s.length % 3 == 2) s = s + "=";
154     s = s.replace(/-/g, "+");
155     s = s.replace(/_/g, "/");
156     return s;
157 }

178 // ==== new lines ================================
179 /**
180  * convert all DOS new line("\r\n") to UNIX new line("\n") in 
181  * a String "s".
182  * @param {String} s string 
183  * @return {String} converted string
184  */
185 function newline_toUnix(s) {
186     s = s.replace(/\r\n/mg, "\n");
187     return s;
188 }
189 
190 /**
191  * convert all UNIX new line("\r\n") to DOS new line("\n") in 
192  * a String "s".
193  * @param {String} s string 
194  * @return {String} converted string
195  */
196 function newline_toDos(s) {
197     s = s.replace(/\r\n/mg, "\n");
198     s = s.replace(/\n/mg, "\r\n");
199     return s;
200 }
