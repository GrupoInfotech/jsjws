 29 /**
 30  * JSON Web Signature(JWS) class.<br/>
 31  * @class JSON Web Signature(JWS) class
 32  * @property {Dictionary} parsedJWS This property is set after JWS signature verification. <br/>
 33  *           Following "parsedJWS_*" properties can be accessed as "parsedJWS.*" because of
 34  *           JsDoc restriction.
 35  * @property {String} parsedJWS_headB64U string of Encrypted JWS Header
 36  * @property {String} parsedJWS_payloadB64U string of Encrypted JWS Payload
 37  * @property {String} parsedJWS_sigvalB64U string of Encrypted JWS signature value
 38  * @property {String} parsedJWS_si string of Signature Input
 39  * @property {String} parsedJWS_sigvalH hexadecimal string of JWS signature value
 40  * @property {String} parsedJWS_sigvalBI BigInteger(defined in jsbn.js) object of JWS signature value
 41  * @property {String} parsedJWS_headS string of decoded JWS Header
 42  * @property {String} parsedJWS_headS string of decoded JWS Payload
 43  * @author Kenji Urushima
 44  * @version 1.1 (07 May 2012)
 45  * @requires base64x.js, json-sans-eval.js and jsrsasign library
 46  * @see <a href="http://kjur.github.com/jsjws/">'jwjws'(JWS JavaScript Library) home page http://kjur.github.com/jsjws/</a>
 47  * @see <a href="http://kjur.github.com/jsrsasigns/">'jwrsasign'(RSA Sign JavaScript Library) home page http://kjur.github.com/jsrsasign/</a>
 48  */
 49 KJUR.jws.JWS = function() {

 51     // === utility =============================================================
 52     /**
 53      * check whether a String "s" is a safe JSON string or not.<br/>
 54      * If a String "s" is a malformed JSON string or an other object type
 55      * this returns 0, otherwise this returns 1.
 56      * @name isSafeJSONString
 57      * @memberOf KJUR.jws.JWS
 58      * @function
 59      * @param {String} s JSON string
 60      * @return {Number} 1 or 0
 61      */
 62     this.isSafeJSONString = function(s, h, p) {
 63 	var o = null;
 64 	try {
 65 	    o = jsonParse(s);
 66 	    if (typeof o != "object") return 0;
 67 	    if (o.constructor === Array) return 0;
 68 	    if (h) h[p] = o;
 69 	    return 1;
 70 	} catch (ex) {
 71 	    return 0;
 72 	}
 73     };
 74 
 75     /**
 76      * read a String "s" as JSON object if it is safe.<br/>
 77      * If a String "s" is a malformed JSON string or not JSON string,
 78      * this returns null, otherwise returns JSON object.
 79      * @name readSafeJSONString
 80      * @memberOf KJUR.jws.JWS
 81      * @function
 82      * @param {String} s JSON string
 83      * @return {Object} JSON object or null
 84      * @since 1.1.1
 85      */
 86     this.readSafeJSONString = function(s) {
 87 	var o = null;
 88 	try {
 89 	    o = jsonParse(s);
 90 	    if (typeof o != "object") return null;
 91 	    if (o.constructor === Array) return null;
 92 	    return o;
 93 	} catch (ex) {
 94 	    return null;
 95 	}
 96     };
 97 
 98     /**
 99      * get Encoed Signature Value from JWS string.<br/>
100      * @name getEncodedSignatureValueFromJWS
101      * @memberOf KJUR.jws.JWS
102      * @function
103      * @param {String} sJWS JWS signature string to be verified
104      * @return {String} string of Encoded Signature Value 
105      * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
106      */
107     this.getEncodedSignatureValueFromJWS = function(sJWS) {
108 	if (sJWS.match(/^[^.]+\.[^.]+\.([^.]+)$/) == null) {
109 	    throw "JWS signature is not a form of 'Head.Payload.SigValue'.";
110 	}
111 	return RegExp.$1;
112     };
113 
114     /**
115      * parse JWS string and set public property 'parsedJWS' dictionary.<br/>
116      * @name parseJWS
117      * @memberOf KJUR.jws.JWS
118      * @function
119      * @param {String} sJWS JWS signature string to be parsed.
120      * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
121      * @throws if JWS Header is a malformed JSON string.
122      * @since 1.1
123      */
124     this.parseJWS = function(sJWS, sigValNotNeeded) {
125 	if ((this.parsedJWS !== undefined) &&
126 	    (sigValNotNeeded || (this.parsedJWS.sigvalH !== undefined))) {
127 	    return;
128 	}
129 	if (sJWS.match(/^([^.]+)\.([^.]+)\.([^.]+)$/) == null) {
130 	    throw "JWS signature is not a form of 'Head.Payload.SigValue'.";
131 	}
132 	var b6Head = RegExp.$1;
133 	var b6Payload = RegExp.$2;
134 	var b6SigVal = RegExp.$3;
135 	var sSI = b6Head + "." + b6Payload;
136 	this.parsedJWS = {};
137 	this.parsedJWS.headB64U = b6Head;
138 	this.parsedJWS.payloadB64U = b6Payload;
139 	this.parsedJWS.sigvalB64U = b6SigVal;
140 	this.parsedJWS.si = sSI;
141 
142 	if (!sigValNotNeeded) {
143 	    var hSigVal = b64utohex(b6SigVal);
144 	    var biSigVal = parseBigInt(hSigVal, 16);
145 	    this.parsedJWS.sigvalH = hSigVal;
146 	    this.parsedJWS.sigvalBI = biSigVal;
147 	}
148 
149 	var sHead = b64utoutf8(b6Head);
150 	var sPayload = b64utoutf8(b6Payload);
151 	this.parsedJWS.headS = sHead;
152 	this.parsedJWS.payloadS = sPayload;
153 
154 	if (! this.isSafeJSONString(sHead, this.parsedJWS, 'headP'))
155 	    throw "malformed JSON string for JWS Head: " + sHead;
156     };

158     // ==== JWS Validation =========================================================
159     function _getSignatureInputByString(sHead, sPayload) {
160 	return utf8tob64u(sHead) + "." + utf8tob64u(sPayload);
161     };
162 
163     function _getHashBySignatureInput(sSignatureInput, sHashAlg) {
164 	var hashfunc = _RSASIGN_HASHHEXFUNC[sHashAlg];
165 	if (hashfunc == null) throw "hash function not defined in jsrsasign: " + sHashAlg;
166 	return hashfunc(sSignatureInput);
167     };
168 
169     function _jws_verifySignature(sHead, sPayload, hSig, hN, hE) {
170 	var sSignatureInput = _getSignatureInputByString(sHead, sPayload);
171 	var biSig = parseBigInt(hSig, 16);
172 	return _rsasign_verifySignatureWithArgs(sSignatureInput, biSig, hN, hE);
173     };
174 
175     /**
176      * verify JWS signature with naked RSA public key.<br/>
177      * This only supports "RS256" and "RS512" algorithm.
178      * @name verifyJWSByNE
179      * @memberOf KJUR.jws.JWS
180      * @function
181      * @param {String} sJWS JWS signature string to be verified
182      * @param {String} hN hexadecimal string for modulus of RSA public key
183      * @param {String} hE hexadecimal string for public exponent of RSA public key
184      * @return {String} returns 1 when JWS signature is valid, otherwise returns 0
185      * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
186      * @throws if JWS Header is a malformed JSON string.
187      */
188     this.verifyJWSByNE = function(sJWS, hN, hE) {
189 	this.parseJWS(sJWS);
190 	return _rsasign_verifySignatureWithArgs(this.parsedJWS.si, this.parsedJWS.sigvalBI, hN, hE);    
191     };
192 
193     /**
194      * verify JWS signature with RSA public key.<br/>
195      * This only supports "RS256", "RS512", "PS256" and "PS512" algorithms.
196      * @name verifyJWSByKey
197      * @memberOf KJUR.jws.JWS
198      * @function
199      * @param {String} sJWS JWS signature string to be verified
200      * @param {RSAKey} RSA public key
201      * @return {Boolean} returns true when JWS signature is valid, otherwise returns false
202      * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
203      * @throws if JWS Header is a malformed JSON string.
204      */
205     this.verifyJWSByKey = function(sJWS, key) {
206 	this.parseJWS(sJWS);
207 	var hashAlg = _jws_getHashAlgFromParsedHead(this.parsedJWS.headP);
208 
209 	if (key.hashAndVerify) {
210 	    return key.hashAndVerify(hashAlg,
211 				     new Buffer(this.parsedJWS.si, 'utf8').toString('base64'),
212 				     b64utob64(this.parsedJWS.sigvalB64U),
213 				     'base64');
214 	} else if (this.parsedJWS.headP['alg'].substr(0, 2) == "PS") {
215 	    return key.verifyStringPSS(this.parsedJWS.si,
216 				       this.parsedJWS.sigvalBI, hashAlg);
217 	} else {
218 	    return key.verifyString(this.parsedJWS.si,
219 				    this.parsedJWS.sigvalBI);
220 	}
221     };
222 
223     /**
224      * verify JWS signature by PEM formatted X.509 certificate.<br/>
225      * This only supports "RS256" and "RS512" algorithm.
226      * @name verifyJWSByPemX509Cert
227      * @memberOf KJUR.jws.JWS
228      * @function
229      * @param {String} sJWS JWS signature string to be verified
230      * @param {String} sPemX509Cert string of PEM formatted X.509 certificate
231      * @return {String} returns 1 when JWS signature is valid, otherwise returns 0
232      * @throws if sJWS is not comma separated string such like "Header.Payload.Signature".
233      * @throws if JWS Header is a malformed JSON string.
234      * @since 1.1
235      */
236     this.verifyJWSByPemX509Cert = function(sJWS, sPemX509Cert) {
237 	this.parseJWS(sJWS);
238 	var x509 = new X509();
239 	x509.readCertPEM(sPemX509Cert);
240 	return x509.subjectPublicKeyRSA.verifyString(this.parsedJWS.si, this.parsedJWS.sigvalH);
241     };

243     // ==== JWS Generation =========================================================
244     function _jws_getHashAlgFromParsedHead(head) {
245 	var sigAlg = head["alg"];
246 	var hashAlg = "";
247 
248 	if (sigAlg != "RS256" && sigAlg != "RS512" &&
249 	    sigAlg != "PS256" && sigAlg != "PS512")
250 	    throw "JWS signature algorithm not supported: " + sigAlg;
251 	if (sigAlg.substr(2) == "256") hashAlg = "sha256";
252 	if (sigAlg.substr(2) == "512") hashAlg = "sha512";
253 	return hashAlg;
254     };
255 
256     function _jws_getHashAlgFromHead(sHead) {
257 	return _jws_getHashAlgFromParsedHead(jsonParse(sHead));
258     };
259 
260     function _jws_generateSignatureValueBySI_NED(sHead, sPayload, sSI, hN, hE, hD) {
261 	var rsa = new RSAKey();
262 	rsa.setPrivate(hN, hE, hD);
263 
264 	var hashAlg = _jws_getHashAlgFromHead(sHead);
265 	var sigValue = rsa.signString(sSI, hashAlg);
266 	return sigValue;
267     };
268 
269     function _jws_generateSignatureValueBySI_Key(sHead, sPayload, sSI, key, head) {
270 	var hashAlg = null;
271 	if (typeof head == "undefined") {
272 	    hashAlg = _jws_getHashAlgFromHead(sHead);
273 	} else {
274 	    hashAlg = _jws_getHashAlgFromParsedHead(head);
275 	}
276 
277 	if (key.hashAndSign) {
278 	    return b64tob64u(key.hashAndSign(hashAlg, sSI, 'binary', 'base64'));
279 	} else if (hashAlg.substr(0, 2) == "PS") {
280 	    return hextob64u(key.signStringPSS(sSI, hashAlg));
281 	} else {
282 	    return hextob64u(key.signString(sSI, hashAlg));
283 	}
284     };
285 
286     function _jws_generateSignatureValueByNED(sHead, sPayload, hN, hE, hD) {
287 	var sSI = _getSignatureInputByString(sHead, sPayload);
288 	return _jws_generateSignatureValueBySI_NED(sHead, sPayload, sSI, hN, hE, hD);
289     };

291     /**
292      * generate JWS signature by Header, Payload and a naked RSA private key.<br/>
293      * This only supports "RS256" and "RS512" algorithm.
294      * @name generateJWSByNED
295      * @memberOf KJUR.jws.JWS
296      * @function
297      * @param {String} sHead string of JWS Header
298      * @param {String} sPayload string of JWS Payload
299      * @param {String} hN hexadecimal string for modulus of RSA public key
300      * @param {String} hE hexadecimal string for public exponent of RSA public key
301      * @param {String} hD hexadecimal string for private exponent of RSA private key
302      * @return {String} JWS signature string
303      * @throws if sHead is a malformed JSON string.
304      * @throws if supported signature algorithm was not specified in JSON Header.
305      */
306     this.generateJWSByNED = function(sHead, sPayload, hN, hE, hD) {
307 	if (! this.isSafeJSONString(sHead)) throw "JWS Head is not safe JSON string: " + sHead;
308 	var sSI = _getSignatureInputByString(sHead, sPayload);
309 	var hSigValue = _jws_generateSignatureValueBySI_NED(sHead, sPayload, sSI, hN, hE, hD);
310 	var b64SigValue = hextob64u(hSigValue);
311 	
312 	this.parsedJWS = {};
313 	this.parsedJWS.headB64U = sSI.split(".")[0];
314 	this.parsedJWS.payloadB64U = sSI.split(".")[1];
315 	this.parsedJWS.sigvalB64U = b64SigValue;
316 
317 	return sSI + "." + b64SigValue;
318     };

320     /**
321      * generate JWS signature by Header, Payload and a RSA private key.<br/>
322      * This only supports "RS256", "RS512", "PS256" and "PS512" algorithms.
323      * @name generateJWSByKey
324      * @memberOf KJUR.jws.JWS
325      * @function
326      * @param {String} sHead string of JWS Header
327      * @param {String} sPayload string of JWS Payload
328      * @param {RSAKey} RSA private key
329      * @return {String} JWS signature string
330      * @throws if sHead is a malformed JSON string.
331      * @throws if supported signature algorithm was not specified in JSON Header.
332      */
333     this.generateJWSByKey = function(sHead, sPayload, key) {
334 	var obj = {};
335 	if (!this.isSafeJSONString(sHead, obj, 'headP'))
336 	    throw "JWS Head is not safe JSON string: " + sHead;
337 	var sSI = _getSignatureInputByString(sHead, sPayload);
338 	var b64SigValue = _jws_generateSignatureValueBySI_Key(sHead, sPayload, sSI, key, obj.headP);
339 
340 	this.parsedJWS = {};
341 	this.parsedJWS.headB64U = sSI.split(".")[0];
342 	this.parsedJWS.payloadB64U = sSI.split(".")[1];
343 	this.parsedJWS.sigvalB64U = b64SigValue;
344 
345 	return sSI + "." + b64SigValue;
346     };

348     // === sign with PKCS#1 RSA private key =====================================================
349     function _jws_generateSignatureValueBySI_PemPrvKey(sHead, sPayload, sSI, sPemPrvKey) {
350 	var rsa = new RSAKey();
351 	rsa.readPrivateKeyFromPEMString(sPemPrvKey);
352 	var hashAlg = _jws_getHashAlgFromHead(sHead);
353 	var sigValue = rsa.signString(sSI, hashAlg);
354 	return sigValue;
355     };

357     /**
358      * generate JWS signature by Header, Payload and a PEM formatted PKCS#1 RSA private key.<br/>
359      * This only supports "RS256" and "RS512" algorithm.
360      * @name generateJWSByP1PrvKey
361      * @memberOf KJUR.jws.JWS
362      * @function
363      * @param {String} sHead string of JWS Header
364      * @param {String} sPayload string of JWS Payload
365      * @param {String} string for sPemPrvKey PEM formatted PKCS#1 RSA private key<br/>
366      *                 Heading and trailing space characters in PEM key will be ignored.
367      * @return {String} JWS signature string
368      * @throws if sHead is a malformed JSON string.
369      * @throws if supported signature algorithm was not specified in JSON Header.
370      * @since 1.1
371      */
372     this.generateJWSByP1PrvKey = function(sHead, sPayload, sPemPrvKey) {
373 	if (! this.isSafeJSONString(sHead)) throw "JWS Head is not safe JSON string: " + sHead;
374 	var sSI = _getSignatureInputByString(sHead, sPayload);
375 	var hSigValue = _jws_generateSignatureValueBySI_PemPrvKey(sHead, sPayload, sSI, sPemPrvKey);
376 	var b64SigValue = hextob64u(hSigValue);
377 
378 	this.parsedJWS = {};
379 	this.parsedJWS.headB64U = sSI.split(".")[0];
380 	this.parsedJWS.payloadB64U = sSI.split(".")[1];
381 	this.parsedJWS.sigvalB64U = b64SigValue;
382 
383 	return sSI + "." + b64SigValue;
384     };
